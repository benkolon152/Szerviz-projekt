function hashText(text) {
  let hash = 0;
 
  for (let index = 0; index < text.length; index += 1) {
    hash = (hash * 31 + text.charCodeAt(index)) >>> 0;
  }
 
  return hash;
}
 
function clamp(value, min, max) {
  return Math.max(min, Math.min(max, value));
}
 
function parseNumber(name, regex, fallback) {
  const match = name.match(regex);
 
  return match ? Number(match[1]) : fallback;
}
 
function parseCapacityGb(name, fallback = 1000) {
  const tb = name.match(/(\d+)\s*TB/i);
  if (tb) return Number(tb[1]) * 1000;
 
  const gb = name.match(/(\d+)\s*GB/i);
  if (gb) return Number(gb[1]);
 
  return fallback;
}

  const noise = hashText(name) % 7000;

    else if (/27|32|34/i.test(name)) base = 120000;
    if (/240|360|QD-OLED/i.test(name)) base += 90000;

  if (category === "keyboards") {
    let base = 18000;
    if (/Azoth|Q6|Q60|Q1|G915|BlackWidow\s*V4|Huntsman|K100/i.test(name)) base = 65000;
    else if (/K70|Apex\s*Pro|MX\s*Mechanical|Alloy\s*Rise/i.test(name)) base = 42000;
    return base + noise;
  }
 
  if (category === "mice") {
    let base = 12000;
    if (/Superlight|V3\s*Pro|V2\s*Pro|Naga\s*V2|Finalmouse|Harpe\s*Ace/i.test(name)) base = 52000;
    else if (/MX\s*Master|G502\s*X|Basilisk\s*V3\s*Pro|Aerox\s*5/i.test(name)) base = 34000;

    if (/TrueNAS|Proxmox/i.test(name)) return 45000 + noise;
    return 0;
  }
 
  return 10000 + noise;
